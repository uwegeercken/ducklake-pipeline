services:

  # ── PostgreSQL (DuckLake Katalog) ─────────────────────────────────────────
  postgres:
    image: postgres:16-alpine
    container_name: postgres
    ports:
      - "5432:5432"
    environment:
      POSTGRES_DB: dl1
      POSTGRES_USER: ducklake
      POSTGRES_PASSWORD: ducklake_secret
    volumes:
      - postgres-data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ducklake -d dl1"]
      interval: 5s
      timeout: 5s
      retries: 10

  # ── pgAdmin ───────────────────────────────────────────────────────────────
  pgadmin:
    image: dpage/pgadmin4:latest
    container_name: pgadmin
    ports:
      - "5050:80"
    environment:
      PGADMIN_DEFAULT_EMAIL: admin@ducklake.com
      PGADMIN_DEFAULT_PASSWORD: admin
    depends_on:
      postgres:
        condition: service_healthy

  # ── Kafka (KRaft, kein Zookeeper) ─────────────────────────────────────────
  # bitnami/kafka: Multi-Arch (ARM64 + AMD64), KRaft nativ unterstützt,
  # kein separater Init-Container notwendig.
  kafka:
    image: bitnami/kafka:latest
    container_name: kafka
    hostname: kafka
    ports:
      - "9092:9092"
    environment:
      # KRaft
      KAFKA_CFG_NODE_ID: 1
      KAFKA_CFG_PROCESS_ROLES: broker,controller
      KAFKA_CFG_CONTROLLER_QUORUM_VOTERS: 1@kafka:9093
      # Listeners – bitnami trennt intern INTERNAL/EXTERNAL/CONTROLLER
      KAFKA_CFG_LISTENERS: PLAINTEXT://:9092,CONTROLLER://:9093
      KAFKA_CFG_ADVERTISED_LISTENERS: PLAINTEXT://kafka:9092
      KAFKA_CFG_LISTENER_SECURITY_PROTOCOL_MAP: CONTROLLER:PLAINTEXT,PLAINTEXT:PLAINTEXT
      KAFKA_CFG_CONTROLLER_LISTENER_NAMES: CONTROLLER
      KAFKA_CFG_INTER_BROKER_LISTENER_NAME: PLAINTEXT
      # Replikation (single node)
      KAFKA_CFG_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
      KAFKA_CFG_TRANSACTION_STATE_LOG_REPLICATION_FACTOR: 1
      KAFKA_CFG_TRANSACTION_STATE_LOG_MIN_ISR: 1
      KAFKA_CFG_AUTO_CREATE_TOPICS_ENABLE: "true"
      # bitnami generiert Cluster-ID automatisch – kein Init-Container nötig
      KAFKA_KRAFT_CLUSTER_ID: "MkU3OEVBNTcwNTJENDM2Qg"
    volumes:
      - kafka-data:/bitnami/kafka
    healthcheck:
      test: ["CMD-SHELL", "kafka-topics.sh --bootstrap-server localhost:9092 --list"]
      interval: 10s
      timeout: 10s
      retries: 10
      start_period: 15s

  # ── Kafka UI ──────────────────────────────────────────────────────────────
  kafka-ui:
    image: ghcr.io/kafbat/kafka-ui:latest
    container_name: kafka-ui
    ports:
      - "8080:8080"
    environment:
      KAFKA_CLUSTERS_0_NAME: ducklake-cluster
      KAFKA_CLUSTERS_0_BOOTSTRAPSERVERS: kafka:9092
    depends_on:
      kafka:
        condition: service_healthy

  # ── MinIO (DuckLake Datenspeicher) ────────────────────────────────────────
  minio:
    image: minio/minio:latest
    container_name: minio
    ports:
      - "9000:9000"
      - "9001:9001"
    environment:
      MINIO_ROOT_USER: minioadmin
      MINIO_ROOT_PASSWORD: minioadmin_secret
    command: server /data --console-address ":9001"
    volumes:
      - minio-data:/data
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9000/minio/health/live"]
      interval: 5s
      timeout: 5s
      retries: 10

  # ── MinIO Bucket Init ─────────────────────────────────────────────────────
  minio-init:
    image: minio/mc:latest
    container_name: minio-init
    depends_on:
      minio:
        condition: service_healthy
    entrypoint: >
      /bin/sh -c "
        mc alias set local http://minio:9000 minioadmin minioadmin_secret &&
        mc mb --ignore-existing local/ducklake1 &&
        echo 'Bucket ducklake1 ready'
      "

  # ── Service 1: ducklake-initializer ──────────────────────────────────────
  ducklake-initializer:
    build:
      context: .
      dockerfile: ducklake-initializer/Dockerfile
    container_name: ducklake-initializer
    depends_on:
      postgres:
        condition: service_healthy
      minio-init:
        condition: service_completed_successfully
    environment:
      POSTGRES_HOST: postgres
      POSTGRES_PORT: 5432
      POSTGRES_DATABASE: dl1
      POSTGRES_USER: ducklake
      POSTGRES_PASSWORD: ducklake_secret
      MINIO_ENDPOINT: http://minio:9000
      MINIO_ACCESS_KEY: minioadmin
      MINIO_SECRET_KEY: minioadmin_secret
      MINIO_BUCKET: ducklake1
      DUCKLAKE_NAME: ducklake1
      DUCKLAKE_SCHEMA: schema1
      DUCKLAKE_TABLE: geoname
    restart: on-failure

  # ── Service 2: data-reader ────────────────────────────────────────────────
  data-reader:
    build:
      context: .
      dockerfile: data-reader/Dockerfile
    container_name: data-reader
    depends_on:
      kafka:
        condition: service_healthy
      ducklake-initializer:
        condition: service_completed_successfully
    environment:
      KAFKA_BOOTSTRAP_SERVERS: kafka:9092
      KAFKA_TOPIC: geoname-events
      DATA_FILE_PATH: ./data/geonames.json
    volumes:
      - ./data:/data:ro
    restart: on-failure

  # ── Service 3: data-writer ────────────────────────────────────────────────
  data-writer:
    build:
      context: .
      dockerfile: data-writer/Dockerfile
    container_name: data-writer
    depends_on:
      kafka:
        condition: service_healthy
      ducklake-initializer:
        condition: service_completed_successfully
    environment:
      KAFKA_BOOTSTRAP_SERVERS: kafka:9092
      KAFKA_TOPIC: geoname-events
      KAFKA_GROUP_ID: data-writer-group
      POSTGRES_HOST: postgres
      POSTGRES_PORT: 5432
      POSTGRES_DATABASE: dl1
      POSTGRES_USER: ducklake
      POSTGRES_PASSWORD: ducklake_secret
      MINIO_ENDPOINT: http://minio:9000
      MINIO_ACCESS_KEY: minioadmin
      MINIO_SECRET_KEY: minioadmin_secret
      MINIO_BUCKET: ducklake1
      DUCKLAKE_NAME: ducklake1
      DUCKLAKE_SCHEMA: schema1
      DUCKLAKE_TABLE: geoname
      BATCH_SIZE: "100"
    restart: unless-stopped

volumes:
  postgres-data:
  kafka-data:
  minio-data:
